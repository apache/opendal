18pub fn convert_error(opendal_error: opendal::Error) -> dav_server::fs::FsError {
19    match opendal_error.kind() {
20        opendal::ErrorKind::AlreadyExists | opendal::ErrorKind::IsSameFile => {
21            dav_server::fs::FsError::Exists
22        }
23        opendal::ErrorKind::NotFound => dav_server::fs::FsError::NotFound,
24        _ => dav_server::fs::FsError::GeneralFailure,
25    }
26}
