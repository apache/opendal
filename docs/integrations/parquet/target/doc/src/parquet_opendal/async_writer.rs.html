18use bytes::Bytes;
19use parquet::arrow::async_writer::AsyncFileWriter;
20use parquet::errors::{ParquetError, Result};
21
22use futures::future::BoxFuture;
23use opendal::Writer;
24
25/// AsyncWriter implements AsyncFileWriter trait by using opendal.
26///
27/// ```no_run
28/// use std::sync::Arc;
29///
30/// use arrow::array::{ArrayRef, Int64Array, RecordBatch};
31///
32/// use opendal::{services::S3Config, Operator};
33/// use parquet::arrow::{arrow_reader::ParquetRecordBatchReaderBuilder, AsyncArrowWriter};
34/// use parquet_opendal::AsyncWriter;
35///
36/// #[tokio::main]
37/// async fn main() {
38///     let mut cfg = S3Config::default();
39///     cfg.access_key_id = Some("my_access_key".to_string());
40///     cfg.secret_access_key = Some("my_secret_key".to_string());
41///     cfg.endpoint = Some("my_endpoint".to_string());
42///     cfg.region = Some("my_region".to_string());
43///     cfg.bucket = "my_bucket".to_string();
44///
45///     // Create a new operator
46///     let operator = Operator::from_config(cfg).unwrap();
47///     let path = "/path/to/file.parquet";
48///
49///     // Create an async writer
50///     let writer = AsyncWriter::new(
51///         operator
52///             .writer_with(path)
53///             .chunk(32 * 1024 * 1024)
54///             .concurrent(8)
55///             .await
56///             .unwrap(),
57///     );
58///
59///     let col = Arc::new(Int64Array::from_iter_values([1, 2, 3])) as ArrayRef;
60///     let to_write = RecordBatch::try_from_iter([("col", col)]).unwrap();
61///     let mut writer = AsyncArrowWriter::try_new(writer, to_write.schema(), None).unwrap();
62///     writer.write(&to_write).await.unwrap();
63///     writer.close().await.unwrap();
64///
65///     let buffer = operator.read(path).await.unwrap().to_bytes();
66///     let mut reader = ParquetRecordBatchReaderBuilder::try_new(buffer)
67///         .unwrap()
68///         .build()
69///         .unwrap();
70///     let read = reader.next().unwrap().unwrap();
71///     assert_eq!(to_write, read);
72/// }
73/// ```
74pub struct AsyncWriter {
75    inner: Writer,
76}
77
78impl AsyncWriter {
79    /// Create a [`AsyncWriter`] by given [`Writer`].
80    pub fn new(writer: Writer) -> Self {
81        Self { inner: writer }
82    }
83}
84
85impl AsyncFileWriter for AsyncWriter {
86    fn write(&mut self, bs: Bytes) -> BoxFuture<'_, Result<()>> {
87        Box::pin(async move {
88            self.inner
89                .write(bs)
90                .await
91                .map_err(|err| ParquetError::External(Box::new(err)))
92        })
93    }
94
95    fn complete(&mut self) -> BoxFuture<'_, Result<()>> {
96        Box::pin(async move {
97            self.inner
98                .close()
99                .await
100                .map(|_| ())
101                .map_err(|err| ParquetError::External(Box::new(err)))
102        })
103    }
104}

106#[cfg(test)]
107mod tests {
108    use std::sync::Arc;
109
110    use super::*;
111    use arrow::array::{ArrayRef, Int64Array, RecordBatch};
112    use opendal::{Operator, services};
113    use parquet::arrow::{AsyncArrowWriter, arrow_reader::ParquetRecordBatchReaderBuilder};
114
115    #[tokio::test]
116    async fn test_basic() {
117        let op = Operator::new(services::Memory::default()).unwrap();
118        let path = "data/test.txt";
119        let mut writer = AsyncWriter::new(op.writer(path).await.unwrap());
120        let bytes = Bytes::from_static(b"hello, world!");
121        writer.write(bytes).await.unwrap();
122        let bytes = Bytes::from_static(b"hello, OpenDAL!");
123        writer.write(bytes).await.unwrap();
124        writer.complete().await.unwrap();
125
126        let bytes = op.read(path).await.unwrap().to_vec();
127        assert_eq!(bytes, b"hello, world!hello, OpenDAL!");
128    }
129
130    #[tokio::test]
131    async fn test_abort() {
132        let op = Operator::new(services::Memory::default()).unwrap();
133        let path = "data/test.txt";
134        let mut writer = AsyncWriter::new(op.writer(path).await.unwrap());
135        let bytes = Bytes::from_static(b"hello, world!");
136        writer.write(bytes).await.unwrap();
137        let bytes = Bytes::from_static(b"hello, OpenDAL!");
138        writer.write(bytes).await.unwrap();
139        drop(writer);
140
141        let exist = op.exists(path).await.unwrap();
142        assert!(!exist);
143    }
144
145    #[tokio::test]
146    async fn test_async_writer() {
147        let operator = Operator::new(services::Memory::default()).unwrap();
148        let path = "/path/to/file.parquet";
149
150        let writer = AsyncWriter::new(
151            operator
152                .writer_with(path)
153                .chunk(32 * 1024 * 1024)
154                .concurrent(8)
155                .await
156                .unwrap(),
157        );
158
159        let col = Arc::new(Int64Array::from_iter_values([1, 2, 3])) as ArrayRef;
160        let to_write = RecordBatch::try_from_iter([("col", col)]).unwrap();
161        let mut writer = AsyncArrowWriter::try_new(writer, to_write.schema(), None).unwrap();
162        writer.write(&to_write).await.unwrap();
163        writer.close().await.unwrap();
164
165        let buffer = operator.read(path).await.unwrap().to_bytes();
166        let mut reader = ParquetRecordBatchReaderBuilder::try_new(buffer)
167            .unwrap()
168            .build()
169            .unwrap();
170        let read = reader.next().unwrap().unwrap();
171        assert_eq!(to_write, read);
172    }
173}
