18use std::sync::Arc;
19
20use futures::{StreamExt, stream::BoxStream};
21use object_store::path::Path as ObjectStorePath;
22use object_store::{ObjectMeta, ObjectStore};
23use opendal::raw::*;
24use opendal::*;
25
26use super::error::parse_error;
27use crate::service::core::format_metadata;
28
29pub struct ObjectStoreLister {
30    stream: BoxStream<'static, object_store::Result<ObjectMeta>>,
31}
32
33impl ObjectStoreLister {
34    pub(crate) async fn new(
35        store: Arc<dyn ObjectStore + 'static>,
36        path: &str,
37        args: OpList,
38    ) -> Result<Self> {
39        // If start_after is specified, use list_with_offset
40        let mut stream = if let Some(start_after) = args.start_after() {
41            store
42                .list_with_offset(
43                    Some(&ObjectStorePath::from(path)),
44                    &ObjectStorePath::from(start_after),
45                )
46                .boxed()
47        } else {
48            store.list(Some(&ObjectStorePath::from(path))).boxed()
49        };
50
51        // Process listing arguments
52        if let Some(limit) = args.limit() {
53            stream = stream.take(limit).boxed();
54        }
55
56        Ok(Self { stream })
57    }
58}
59
60// ObjectStoreLister is safe to share between threads, because it only has &mut self API calls.
61unsafe impl Sync for ObjectStoreLister {}
62
63impl oio::List for ObjectStoreLister {
64    async fn next(&mut self) -> Result<Option<oio::Entry>> {
65        match self.stream.next().await {
66            Some(Ok(meta)) => {
67                let metadata = format_metadata(&meta);
68                let entry = oio::Entry::new(meta.location.as_ref(), metadata.clone());
69                Ok(Some(entry))
70            }
71            Some(Err(e)) => Err(parse_error(e)),
72            None => Ok(None),
73        }
74    }
75}
