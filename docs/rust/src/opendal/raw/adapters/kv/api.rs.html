29/// Scan is the async iterator returned by `Adapter::scan`.
30pub trait Scan: Send + Sync + Unpin {
31    /// Fetch the next key in the current key prefix
32    ///
33    /// `Ok(None)` means no further key will be returned
34    fn next(&mut self) -> impl Future<Output = Result<Option<String>>> + MaybeSend;
35}
36
37/// A noop implementation of Scan
38impl Scan for () {
39    async fn next(&mut self) -> Result<Option<String>> {
40        Ok(None)
41    }
42}
43
44/// A Scan implementation for all trivial non-async iterators
45pub struct ScanStdIter<I>(I);
46
47#[cfg(any(
48    feature = "services-cloudflare-kv",
49    feature = "services-rocksdb",
50    feature = "services-sled"
51))]
52impl<I> ScanStdIter<I>
53where
54    I: Iterator<Item = Result<String>> + Unpin + Send + Sync,
55{
56    /// Create a new ScanStdIter from an Iterator
57    pub(crate) fn new(inner: I) -> Self {
58        Self(inner)
59    }
60}
61
62impl<I> Scan for ScanStdIter<I>
63where
64    I: Iterator<Item = Result<String>> + Unpin + Send + Sync,
65{
66    async fn next(&mut self) -> Result<Option<String>> {
67        self.0.next().transpose()
68    }
69}
70
71/// A type-erased wrapper of Scan
72pub type Scanner = Box<dyn ScanDyn>;
73
74pub trait ScanDyn: Unpin + Send + Sync {
75    fn next_dyn(&mut self) -> BoxedFuture<'_, Result<Option<String>>>;
76}
77
78impl<T: Scan + ?Sized> ScanDyn for T {
79    fn next_dyn(&mut self) -> BoxedFuture<'_, Result<Option<String>>> {
80        Box::pin(self.next())
81    }
82}
83
84impl<T: ScanDyn + ?Sized> Scan for Box<T> {
85    async fn next(&mut self) -> Result<Option<String>> {
86        self.deref_mut().next_dyn().await
87    }
88}

90/// KvAdapter is the adapter to underlying kv services.
91///
92/// By implement this trait, any kv service can work as an OpenDAL Service.
93pub trait Adapter: Send + Sync + Debug + Unpin + 'static {
94    /// TODO: use default associate type `= ()` after stabilized
95    type Scanner: Scan;
96
97    /// Return the info of this key value accessor.
98    fn info(&self) -> Info;
99
100    /// Get a key from service.
101    ///
102    /// - return `Ok(None)` if this key is not exist.
103    fn get(&self, path: &str) -> impl Future<Output = Result<Option<Buffer>>> + MaybeSend;
104
105    /// Set a key into service.
106    fn set(&self, path: &str, value: Buffer) -> impl Future<Output = Result<()>> + MaybeSend;
107
108    /// Delete a key from service.
109    ///
110    /// - return `Ok(())` even if this key is not exist.
111    fn delete(&self, path: &str) -> impl Future<Output = Result<()>> + MaybeSend;
112
113    /// Scan a key prefix to get all keys that start with this key.
114    fn scan(&self, path: &str) -> impl Future<Output = Result<Self::Scanner>> + MaybeSend {
115        let _ = path;
116
117        ready(Err(Error::new(
118            ErrorKind::Unsupported,
119            "kv adapter doesn't support this operation",
120        )
121        .with_operation("kv::Adapter::scan")))
122    }
123
124    /// Append a key into service
125    fn append(&self, path: &str, value: &[u8]) -> impl Future<Output = Result<()>> + MaybeSend {
126        let _ = path;
127        let _ = value;
128
129        ready(Err(Error::new(
130            ErrorKind::Unsupported,
131            "kv adapter doesn't support this operation",
132        )
133        .with_operation("kv::Adapter::append")))
134    }
135}

137/// Info for this key value accessor.
138pub struct Info {
139    scheme: Scheme,
140    name: String,
141    capabilities: Capability,
142}
143
144impl Info {
145    /// Create a new KeyValueAccessorInfo.
146    pub fn new(scheme: Scheme, name: &str, capabilities: Capability) -> Self {
147        Self {
148            scheme,
149            name: name.to_string(),
150            capabilities,
151        }
152    }
153
154    /// Get the scheme.
155    pub fn scheme(&self) -> Scheme {
156        self.scheme
157    }
158
159    /// Get the name.
160    pub fn name(&self) -> &str {
161        &self.name
162    }
163
164    /// Get the capabilities.
165    pub fn capabilities(&self) -> Capability {
166        self.capabilities
167    }
