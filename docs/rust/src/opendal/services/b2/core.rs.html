75    /// [b2_authorize_account](https://www.backblaze.com/apidocs/b2-authorize-account)
76    pub async fn get_auth_info(&self) -> Result<AuthInfo> {
77        {
78            let signer = self.signer.read().await;
79
80            if !signer.auth_info.authorization_token.is_empty()
81                && signer.auth_info.expires_in > Timestamp::now()
82            {
83                let auth_info = signer.auth_info.clone();
84                return Ok(auth_info);
85            }
86        }
87
88        {
89            let mut signer = self.signer.write().await;
90            let req = Request::get("https://api.backblazeb2.com/b2api/v2/b2_authorize_account")
91                .header(
92                    header::AUTHORIZATION,
93                    format_authorization_by_basic(
94                        &signer.application_key_id,
95                        &signer.application_key,
96                    )?,
97                )
98                .body(Buffer::new())
99                .map_err(new_request_build_error)?;
100
101            let resp = self.info.http_client().send(req).await?;
102            let status = resp.status();
103
104            match status {
105                StatusCode::OK => {
106                    let resp_body = resp.into_body();
107                    let token: AuthorizeAccountResponse =
108                        serde_json::from_reader(resp_body.reader())
109                            .map_err(new_json_deserialize_error)?;
110                    signer.auth_info = AuthInfo {
111                        authorization_token: token.authorization_token.clone(),
112                        api_url: token.api_url.clone(),
113                        download_url: token.download_url.clone(),
114                        // This authorization token is valid for at most 24 hours.
115                        expires_in: Timestamp::now() + Duration::from_secs(20 * 60 * 60),
116                    };
117                }
118                _ => {
119                    return Err(parse_error(resp));
120                }
121            }
122            Ok(signer.auth_info.clone())
123        }
124    }
125}

128    pub async fn download_file_by_name(
129        &self,
130        path: &str,
131        range: BytesRange,
132        _args: &OpRead,
133    ) -> Result<Response<HttpBody>> {
134        let path = build_abs_path(&self.root, path);
135
136        let auth_info = self.get_auth_info().await?;
137
138        // Construct headers to add to the request
139        let url = format!(
140            "{}/file/{}/{}",
141            auth_info.download_url,
142            self.bucket,
143            percent_encode_path(&path)
144        );
145
146        let mut req = Request::get(&url);
147
148        req = req.header(header::AUTHORIZATION, auth_info.authorization_token);
149
150        if !range.is_full() {
151            req = req.header(header::RANGE, range.to_header());
152        }
153
154        let req = req.extension(Operation::Read);
155
156        let req = req.body(Buffer::new()).map_err(new_request_build_error)?;
157
158        self.info.http_client().fetch(req).await
159    }

235    pub async fn upload_file(
236        &self,
237        path: &str,
238        size: Option<u64>,
239        args: &OpWrite,
240        body: Buffer,
241    ) -> Result<Response<Buffer>> {
242        let resp = self.get_upload_url().await?;
243
244        let p = build_abs_path(&self.root, path);
245
246        let mut req = Request::post(resp.upload_url);
247
248        req = req.header(X_BZ_FILE_NAME, percent_encode_path(&p));
249
250        req = req.header(header::AUTHORIZATION, resp.authorization_token);
251
252        req = req.header(X_BZ_CONTENT_SHA1, "do_not_verify");
253
254        if let Some(size) = size {
255            req = req.header(header::CONTENT_LENGTH, size.to_string())
256        }
257
258        if let Some(mime) = args.content_type() {
259            req = req.header(header::CONTENT_TYPE, mime)
260        } else {
261            req = req.header(header::CONTENT_TYPE, "b2/x-auto")
262        }
263
264        if let Some(pos) = args.content_disposition() {
265            req = req.header(header::CONTENT_DISPOSITION, pos)
266        }
267
268        let req = req.extension(Operation::Write);
269
270        // Set body
271        let req = req.body(body).map_err(new_request_build_error)?;
272
273        self.send(req).await
274    }

423    pub async fn get_file_info(&self, path: &str, delimiter: Option<&str>) -> Result<File> {
424        let resp = self
425            .list_file_names_raw(Some(path), delimiter, None, None, Operation::Stat)
426            .await?;
427
428        let status = resp.status();
429        match status {
430            StatusCode::OK => {
431                let bs = resp.into_body();
432                let mut resp: ListFileNamesResponse =
433                    serde_json::from_reader(bs.reader()).map_err(new_json_deserialize_error)?;
434
435                if resp.files.is_empty() {
436                    return Err(Error::new(ErrorKind::NotFound, "no such file or directory"));
437                }
438                Ok(resp.files.swap_remove(0))
439            }
440            _ => Err(parse_error(resp)),
441        }
442    }

455    async fn list_file_names_raw(
456        &self,
457        prefix: Option<&str>,
458        delimiter: Option<&str>,
459        limit: Option<usize>,
460        start_after: Option<String>,
461        operation: Operation,
462    ) -> Result<Response<Buffer>> {
463        let auth_info = self.get_auth_info().await?;
464
465        let url = format!("{}/b2api/v2/b2_list_file_names", auth_info.api_url);
466
467        let mut url = QueryPairsWriter::new(&url);
468        url = url.push("bucketId", &self.bucket_id);
469
470        if let Some(prefix) = prefix {
471            let prefix = build_abs_path(&self.root, prefix);
472            if !prefix.is_empty() {
473                url = url.push("prefix", &percent_encode_path(&prefix));
474            }
475        }
476
477        if let Some(limit) = limit {
478            url = url.push("maxFileCount", &limit.to_string());
479        }
480
481        if let Some(start_after) = start_after {
482            url = url.push("startFileName", &percent_encode_path(&start_after));
483        }
484
485        if let Some(delimiter) = delimiter {
486            url = url.push("delimiter", delimiter);
487        }
488
489        let mut req = Request::get(url.finish());
490
491        req = req.header(header::AUTHORIZATION, auth_info.authorization_token);
492
493        req = req.extension(operation);
494
495        // Set body
496        let req = req.body(Buffer::new()).map_err(new_request_build_error)?;
497
498        self.send(req).await
499    }

560#[derive(Clone)]
561pub struct B2Signer {
562    /// The application_key_id of this core.
563    pub application_key_id: String,
564    /// The application_key of this core.
565    pub application_key: String,
566
567    pub auth_info: AuthInfo,
568}

570#[derive(Clone)]
571pub struct AuthInfo {
572    pub authorization_token: String,
573    /// The base URL to use for all API calls except for uploading and downloading files.
574    pub api_url: String,
575    /// The base URL to use for downloading files.
576    pub download_url: String,
577    /// The time when the authorization token expires.
578    pub expires_in: Timestamp,
579}

582    fn default() -> Self {
583        B2Signer {
584            application_key: String::new(),
585            application_key_id: String::new(),

587            auth_info: AuthInfo {
588                authorization_token: String::new(),
589                api_url: String::new(),
590                download_url: String::new(),
591                expires_in: Timestamp::MIN,
592            },
593        }
594    }

684#[serde(rename_all = "camelCase")]
685pub struct File {
686    pub file_id: Option<String>,
687    pub content_length: u64,
688    pub content_md5: Option<String>,
689    pub content_type: Option<String>,
690    pub file_name: String,
691}
692
693pub(super) fn parse_file_info(file: &File) -> Metadata {
694    if file.file_name.ends_with('/') {
695        return Metadata::new(EntryMode::DIR);
696    }
697
698    let mut metadata = Metadata::new(EntryMode::FILE);
699
700    metadata.set_content_length(file.content_length);
701
702    if let Some(content_md5) = &file.content_md5 {
703        metadata.set_content_md5(content_md5);
704    }
705
706    if let Some(content_type) = &file.content_type {
707        metadata.set_content_type(content_type);
708    }
709
710    metadata
711}
