18use std::fmt::Debug;
19
20use serde::Deserialize;
21use serde::Serialize;
22
23/// Config for GitHub Action Cache Services support.
24#[derive(Default, Debug, Serialize, Deserialize, Clone, PartialEq, Eq)]
25#[serde(default)]
26#[non_exhaustive]
27pub struct GhacConfig {
28    /// The root path for ghac.
29    pub root: Option<String>,
30    /// The version that used by cache.
31    pub version: Option<String>,
32    /// The endpoint for ghac service.
33    pub endpoint: Option<String>,
34    /// The runtime token for ghac service.
35    pub runtime_token: Option<String>,
36}
