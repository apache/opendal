39const DEFAULT_REDIS_ENDPOINT: &str = "tcp://127.0.0.1:6379";
40const DEFAULT_REDIS_PORT: u16 = 6379;

42/// [Redis](https://redis.io/) services support.
43#[doc = include_str!("docs.md")]
44#[derive(Debug, Default)]
45pub struct RedisBuilder {
46    pub(super) config: RedisConfig,
47}
48
49impl RedisBuilder {
50    /// set the network address of redis service.
51    ///
52    /// currently supported schemes:
53    /// - no scheme: will be seen as "tcp"
54    /// - "tcp" or "redis": unsecured redis connections
55    /// - "rediss": secured redis connections
56    /// - "unix" or "redis+unix": unix socket connection
57    pub fn endpoint(mut self, endpoint: &str) -> Self {
58        if !endpoint.is_empty() {
59            self.config.endpoint = Some(endpoint.to_owned());
60        }
61        self
62    }
63
64    /// set the network address of redis cluster service.
65    /// This parameter is mutually exclusive with the endpoint parameter.
66    ///
67    /// currently supported schemes:
68    /// - no scheme: will be seen as "tcp"
69    /// - "tcp" or "redis": unsecured redis connections
70    /// - "rediss": secured redis connections
71    /// - "unix" or "redis+unix": unix socket connection
72    pub fn cluster_endpoints(mut self, cluster_endpoints: &str) -> Self {
73        if !cluster_endpoints.is_empty() {
74            self.config.cluster_endpoints = Some(cluster_endpoints.to_owned());
75        }
76        self
77    }
78
79    /// set the username for redis
80    ///
81    /// default: no username
82    pub fn username(mut self, username: &str) -> Self {
83        if !username.is_empty() {
84            self.config.username = Some(username.to_owned());
85        }
86        self
87    }
88
89    /// set the password for redis
90    ///
91    /// default: no password
92    pub fn password(mut self, password: &str) -> Self {
93        if !password.is_empty() {
94            self.config.password = Some(password.to_owned());
95        }
96        self
97    }
98
99    /// set the db used in redis
100    ///
101    /// default: 0
102    pub fn db(mut self, db: i64) -> Self {
103        self.config.db = db;
104        self
105    }
106
107    /// Set the default ttl for redis services.
108    ///
109    /// If set, we will specify `EX` for write operations.
110    pub fn default_ttl(mut self, ttl: Duration) -> Self {
111        self.config.default_ttl = Some(ttl);
112        self
113    }
114
115    /// set the working directory, all operations will be performed under it.
116    ///
117    /// default: "/"
118    pub fn root(mut self, root: &str) -> Self {
119        self.config.root = if root.is_empty() {
120            None
121        } else {
122            Some(root.to_string())
123        };
124
125        self
126    }
127
128    /// Sets the maximum number of connections managed by the pool.
129    ///
130    /// Defaults to 10.
131    ///
132    /// # Panics
133    ///
134    /// Will panic if `max_size` is 0.
135    #[must_use]
136    pub fn connection_pool_max_size(mut self, max_size: u32) -> Self {
137        assert!(max_size > 0, "max_size must be greater than zero!");
138        self.config.connection_pool_max_size = Some(max_size);
139        self
140    }
141}

143impl Builder for RedisBuilder {
144    type Config = RedisConfig;
145
146    fn build(self) -> Result<impl Access> {
147        let root = normalize_root(
148            self.config
149                .root
150                .clone()
151                .unwrap_or_else(|| "/".to_string())
152                .as_str(),
153        );
154
155        if let Some(endpoints) = self.config.cluster_endpoints.clone() {
156            let mut cluster_endpoints: Vec<ConnectionInfo> = Vec::default();
157            for endpoint in endpoints.split(',') {
158                cluster_endpoints.push(self.get_connection_info(endpoint.to_string())?);
159            }
160            let mut client_builder = ClusterClientBuilder::new(cluster_endpoints);
161            if let Some(username) = &self.config.username {
162                client_builder = client_builder.username(username.clone());
163            }
164            if let Some(password) = &self.config.password {
165                client_builder = client_builder.password(password.clone());
166            }
167            let client = client_builder.build().map_err(format_redis_error)?;
168
169            let conn = OnceCell::new();
170
171            Ok(RedisBackend::new(RedisCore {
172                addr: endpoints,
173                client: None,
174                cluster_client: Some(client),
175                conn,
176                default_ttl: self.config.default_ttl,
177                connection_pool_max_size: self.config.connection_pool_max_size,
178            })
179            .with_normalized_root(root))
180        } else {
181            let endpoint = self
182                .config
183                .endpoint
184                .clone()
185                .unwrap_or_else(|| DEFAULT_REDIS_ENDPOINT.to_string());
186
187            let client =
188                Client::open(self.get_connection_info(endpoint.clone())?).map_err(|e| {
189                    Error::new(ErrorKind::ConfigInvalid, "invalid or unsupported scheme")
190                        .with_context("service", REDIS_SCHEME)
191                        .with_context("endpoint", self.config.endpoint.as_ref().unwrap())
192                        .with_context("db", self.config.db.to_string())
193                        .set_source(e)
194                })?;
195
196            let conn = OnceCell::new();
197            Ok(RedisBackend::new(RedisCore {
198                addr: endpoint,
199                client: Some(client),
200                cluster_client: None,
201                conn,
202                default_ttl: self.config.default_ttl,
203                connection_pool_max_size: self.config.connection_pool_max_size,
204            })
205            .with_normalized_root(root))
206        }
207    }
208}

210impl RedisBuilder {
211    fn get_connection_info(&self, endpoint: String) -> Result<ConnectionInfo> {
212        let ep_url = endpoint.parse::<Uri>().map_err(|e| {
213            Error::new(ErrorKind::ConfigInvalid, "endpoint is invalid")
214                .with_context("service", REDIS_SCHEME)
215                .with_context("endpoint", endpoint)
216                .set_source(e)
217        })?;
218
219        let con_addr = match ep_url.scheme_str() {
220            Some("tcp") | Some("redis") | None => {
221                let host = ep_url
222                    .host()
223                    .map(|h| h.to_string())
224                    .unwrap_or_else(|| "127.0.0.1".to_string());
225                let port = ep_url.port_u16().unwrap_or(DEFAULT_REDIS_PORT);
226                ConnectionAddr::Tcp(host, port)
227            }
228            Some("rediss") => {
229                let host = ep_url
230                    .host()
231                    .map(|h| h.to_string())
232                    .unwrap_or_else(|| "127.0.0.1".to_string());
233                let port = ep_url.port_u16().unwrap_or(DEFAULT_REDIS_PORT);
234                ConnectionAddr::TcpTls {
235                    host,
236                    port,
237                    insecure: false,
238                    tls_params: None,
239                }
240            }
241            Some("unix") | Some("redis+unix") => {
242                let path = PathBuf::from(ep_url.path());
243                ConnectionAddr::Unix(path)
244            }
245            Some(s) => {
246                return Err(
247                    Error::new(ErrorKind::ConfigInvalid, "invalid or unsupported scheme")
248                        .with_context("service", REDIS_SCHEME)
249                        .with_context("scheme", s),
250                );
251            }
252        };
253
254        let redis_info = RedisConnectionInfo {
255            db: self.config.db,
256            username: self.config.username.clone(),
257            password: self.config.password.clone(),
258            protocol: ProtocolVersion::RESP2,
259        };
260
261        Ok(ConnectionInfo {
262            addr: con_addr,
263            redis: redis_info,
264        })
265    }
266}

268/// RedisBackend implements Access trait directly
269#[derive(Debug, Clone)]
270pub struct RedisBackend {
271    core: Arc<RedisCore>,
272    root: String,
273    info: Arc<AccessorInfo>,
274}
275
276impl RedisBackend {
277    fn new(core: RedisCore) -> Self {
278        let info = AccessorInfo::default();
279        info.set_scheme(REDIS_SCHEME);
280        info.set_name(&core.addr);
281        info.set_root("/");
282        info.set_native_capability(Capability {
283            read: true,
284            write: true,
285            delete: true,
286            stat: true,
287            write_can_empty: true,
288            shared: true,
289            ..Default::default()
290        });
291
292        Self {
293            core: Arc::new(core),
294            root: "/".to_string(),
295            info: Arc::new(info),
296        }
297    }
298
299    fn with_normalized_root(mut self, root: String) -> Self {
300        self.info.set_root(&root);
301        self.root = root;
302        self
303    }
304}
305
306impl Access for RedisBackend {
307    type Reader = Buffer;
308    type Writer = RedisWriter;
309    type Lister = ();
310    type Deleter = oio::OneShotDeleter<RedisDeleter>;
311
312    fn info(&self) -> Arc<AccessorInfo> {
313        self.info.clone()
314    }
315
316    async fn stat(&self, path: &str, _: OpStat) -> Result<RpStat> {
317        let p = build_abs_path(&self.root, path);
318
319        if p == build_abs_path(&self.root, "") {
320            Ok(RpStat::new(Metadata::new(EntryMode::DIR)))
321        } else {
322            let bs = self.core.get(&p).await?;
323            match bs {
324                Some(bs) => Ok(RpStat::new(
325                    Metadata::new(EntryMode::FILE).with_content_length(bs.len() as u64),
326                )),
327                None => Err(Error::new(ErrorKind::NotFound, "key not found in redis")),
328            }
329        }
330    }
331
332    async fn read(&self, path: &str, args: OpRead) -> Result<(RpRead, Self::Reader)> {
333        let p = build_abs_path(&self.root, path);
334
335        let range = args.range();
336        let buffer = if range.is_full() {
337            // Full read - use GET
338            match self.core.get(&p).await? {
339                Some(bs) => bs,
340                None => return Err(Error::new(ErrorKind::NotFound, "key not found in redis")),
341            }
342        } else {
343            // Range read - use GETRANGE
344            let start = range.offset() as isize;
345            let end = match range.size() {
346                Some(size) => (range.offset() + size - 1) as isize,
347                None => -1, // Redis uses -1 for end of string
348            };
349
350            match self.core.get_range(&p, start, end).await? {
351                Some(bs) => bs,
352                None => return Err(Error::new(ErrorKind::NotFound, "key not found in redis")),
353            }
354        };
355
356        Ok((RpRead::new(), buffer))
357    }
358
359    async fn write(&self, path: &str, _: OpWrite) -> Result<(RpWrite, Self::Writer)> {
360        let p = build_abs_path(&self.root, path);
361        Ok((RpWrite::new(), RedisWriter::new(self.core.clone(), p)))
362    }
363
364    async fn delete(&self) -> Result<(RpDelete, Self::Deleter)> {
365        Ok((
366            RpDelete::default(),
367            oio::OneShotDeleter::new(RedisDeleter::new(self.core.clone(), self.root.clone())),
368        ))
369    }
