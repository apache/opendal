18use crate::raw::*;
19use crate::*;
20use futures::Sink;
21use std::marker::PhantomData;
22use std::pin::Pin;
23use std::task::{ready, Context, Poll};
24
25/// FuturesDeleteSink is a sink that generated by [`Deleter`]
26pub struct FuturesDeleteSink<T: IntoDeleteInput> {
27    state: State,
28    _phantom: PhantomData<T>,
29}
30
31enum State {
32    Idle(Option<Deleter>),
33    Delete(BoxedStaticFuture<(Deleter, Result<()>)>),
34    Flush(BoxedStaticFuture<(Deleter, Result<usize>)>),
35    Close(BoxedStaticFuture<(Deleter, Result<()>)>),
36}
37
38impl<T: IntoDeleteInput> FuturesDeleteSink<T> {
39    #[inline]
40    pub(super) fn new(deleter: Deleter) -> Self {
41        Self {
42            state: State::Idle(Some(deleter)),
43            _phantom: PhantomData,
44        }
45    }
46}
47
48impl<T: IntoDeleteInput> Sink<T> for FuturesDeleteSink<T> {
49    type Error = Error;
50
51    fn poll_ready(mut self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Result<()>> {
52        match &mut self.state {
53            State::Idle(_) => Poll::Ready(Ok(())),
54            State::Delete(fut) => {
55                let (deleter, res) = ready!(fut.as_mut().poll(cx));
56                self.state = State::Idle(Some(deleter));
57                Poll::Ready(res.map(|_| ()))
58            }
59            State::Flush(fut) => {
60                let (deleter, res) = ready!(fut.as_mut().poll(cx));
61                self.state = State::Idle(Some(deleter));
62                Poll::Ready(res.map(|_| ()))
63            }
64            State::Close(fut) => {
65                let (deleter, res) = ready!(fut.as_mut().poll(cx));
66                self.state = State::Idle(Some(deleter));
67                Poll::Ready(res.map(|_| ()))
68            }
69        }
70    }
71
72    fn start_send(mut self: Pin<&mut Self>, item: T) -> Result<()> {
73        match &mut self.state {
74            State::Idle(deleter) => {
75                let mut deleter = deleter.take().ok_or_else(|| {
76                    Error::new(
77                        ErrorKind::Unexpected,
78                        "FuturesDeleteSink has been closed or errored",
79                    )
80                })?;
81                let input = item.into_delete_input();
82                let fut = async move {
83                    let res = deleter.delete(input).await;
84                    (deleter, res)
85                };
86                self.state = State::Delete(Box::pin(fut));
87                Ok(())
88            }
89            _ => Err(Error::new(
90                ErrorKind::Unexpected,
91                "FuturesDeleteSink is not ready to send, please poll_ready first",
92            )),
93        }
94    }
95
96    fn poll_flush(mut self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Result<()>> {
97        loop {
98            match &mut self.state {
99                State::Idle(deleter) => {
100                    let mut deleter = deleter.take().ok_or_else(|| {
101                        Error::new(
102                            ErrorKind::Unexpected,
103                            "FuturesDeleteSink has been closed or errored",
104                        )
105                    })?;
106                    let fut = async move {
107                        let res = deleter.flush().await;
108                        (deleter, res)
109                    };
110                    self.state = State::Flush(Box::pin(fut));
111                    return Poll::Ready(Ok(()));
112                }
113                State::Delete(fut) => {
114                    let (deleter, res) = ready!(fut.as_mut().poll(cx));
115                    self.state = State::Idle(Some(deleter));
116                    res?;
117                    continue;
118                }
119                State::Flush(fut) => {
120                    let (deleter, res) = ready!(fut.as_mut().poll(cx));
121                    self.state = State::Idle(Some(deleter));
122                    let _ = res?;
123                    return Poll::Ready(Ok(()));
124                }
125                State::Close(fut) => {
126                    let (deleter, res) = ready!(fut.as_mut().poll(cx));
127                    self.state = State::Idle(Some(deleter));
128                    res?;
129                    continue;
130                }
131            };
132        }
133    }
134
135    fn poll_close(mut self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Result<()>> {
136        loop {
137            match &mut self.state {
138                State::Idle(deleter) => {
139                    let mut deleter = deleter.take().ok_or_else(|| {
140                        Error::new(
141                            ErrorKind::Unexpected,
142                            "FuturesDeleteSink has been closed or errored",
143                        )
144                    })?;
145                    let fut = async move {
146                        let res = deleter.close().await;
147                        (deleter, res)
148                    };
149                    self.state = State::Close(Box::pin(fut));
150                    return Poll::Ready(Ok(()));
151                }
152                State::Delete(fut) => {
153                    let (deleter, res) = ready!(fut.as_mut().poll(cx));
154                    self.state = State::Idle(Some(deleter));
155                    res?;
156                    continue;
157                }
158                State::Flush(fut) => {
159                    let (deleter, res) = ready!(fut.as_mut().poll(cx));
160                    self.state = State::Idle(Some(deleter));
161                    res?;
162                    continue;
163                }
164                State::Close(fut) => {
165                    let (deleter, res) = ready!(fut.as_mut().poll(cx));
166                    self.state = State::Idle(Some(deleter));
167                    return Poll::Ready(res);
168                }
169            };
170        }
171    }
172}
