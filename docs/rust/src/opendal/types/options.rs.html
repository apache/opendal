31/// Options for list operations.
32
33#[derive(Debug, Clone, Default, Eq, PartialEq)]
34pub struct ListOptions {
35    /// The limit passed to underlying service to specify the max results
36    /// that could return per-request.
37    ///
38    /// Users could use this to control the memory usage of list operation.
39    pub limit: Option<usize>,
40    /// The start_after passes to underlying service to specify the specified key
41    /// to start listing from.
42    pub start_after: Option<String>,
43    /// The recursive is used to control whether the list operation is recursive.
44    ///
45    /// - If `false`, list operation will only list the entries under the given path.
46    /// - If `true`, list operation will list all entries that starts with given path.
47    ///
48    /// Default to `false`.
49    pub recursive: bool,
50    /// The version is used to control whether the object versions should be returned.
51    ///
52    /// - If `false`, list operation will not return with object versions
53    /// - If `true`, list operation will return with object versions if object versioning is supported
54    ///   by the underlying service
55    ///
56    /// Default to `false`
57    pub versions: bool,
58    /// The deleted is used to control whether the deleted objects should be returned.
59    ///
60    /// - If `false`, list operation will not return with deleted objects
61    /// - If `true`, list operation will return with deleted objects if object versioning is supported
62    ///   by the underlying service
63    ///
64    /// Default to `false`
65    pub deleted: bool,
66}

68/// Options for read operations.
69#[derive(Debug, Clone, Default, Eq, PartialEq)]
70pub struct ReadOptions {
71    /// Set `range` for this operation.
72    ///
73    /// If we have a file with size `n`.
74    ///
75    /// - `..` means read bytes in range `[0, n)` of file.
76    /// - `0..1024` and `..1024` means read bytes in range `[0, 1024)` of file
77    /// - `1024..` means read bytes in range `[1024, n)` of file
78    ///
79    /// The type implements `From<RangeBounds<u64>>`, so users can use `(1024..).into()` instead.
80    pub range: BytesRange,
81    /// Set `version` for this operation.
82    ///
83    /// This option can be used to retrieve the data of a specified version of the given path.
84    ///
85    /// If the version doesn't exist, an error with kind [`ErrorKind::NotFound`] will be returned.
86    pub version: Option<String>,
87
88    /// Set `if_match` for this operation.
89    ///
90    /// This option can be used to check if the file's `ETag` matches the given `ETag`.
91    ///
92    /// If file exists and it's etag doesn't match, an error with kind [`ErrorKind::ConditionNotMatch`]
93    /// will be returned.
94    pub if_match: Option<String>,
95    /// Set `if_none_match` for this operation.
96    ///
97    /// This option can be used to check if the file's `ETag` doesn't match the given `ETag`.
98    ///
99    /// If file exists and it's etag match, an error with kind [`ErrorKind::ConditionNotMatch`]
100    /// will be returned.
101    pub if_none_match: Option<String>,
102    /// Set `if_modified_since` for this operation.
103    ///
104    /// This option can be used to check if the file has been modified since the given timestamp.
105    ///
106    /// If file exists and it hasn't been modified since the specified time, an error with kind
107    /// [`ErrorKind::ConditionNotMatch`] will be returned.
108    pub if_modified_since: Option<Timestamp>,
109    /// Set `if_unmodified_since` for this operation.
110    ///
111    /// This feature can be used to check if the file hasn't been modified since the given timestamp.
112    ///
113    /// If file exists and it has been modified since the specified time, an error with kind
114    /// [`ErrorKind::ConditionNotMatch`] will be returned.
115    pub if_unmodified_since: Option<Timestamp>,
116
117    /// Set `concurrent` for the operation.
118    ///
119    /// OpenDAL by default to read file without concurrent. This is not efficient for cases when users
120    /// read large chunks of data. By setting `concurrent`, opendal will reading files concurrently
121    /// on support storage services.
122    ///
123    /// By setting `concurrent`, opendal will fetch chunks concurrently with
124    /// the give chunk size.
125    ///
126    /// Refer to [`crate::docs::performance`] for more details.
127    pub concurrent: usize,
128    /// Set `chunk` for the operation.
129    ///
130    /// OpenDAL will use services' preferred chunk size by default. Users can set chunk based on their own needs.
131    ///
132    /// Refer to [`crate::docs::performance`] for more details.
133    pub chunk: Option<usize>,
134    /// Controls the optimization strategy for range reads in [`Reader::fetch`].
135    ///
136    /// When performing range reads, if the gap between two requested ranges is smaller than
137    /// the configured `gap` size, OpenDAL will merge these ranges into a single read request
138    /// and discard the unrequested data in between. This helps reduce the number of API calls
139    /// to remote storage services.
140    ///
141    /// This optimization is particularly useful when performing multiple small range reads
142    /// that are close to each other, as it reduces the overhead of multiple network requests
143    /// at the cost of transferring some additional data.
144    ///
145    /// Refer to [`crate::docs::performance`] for more details.
146    pub gap: Option<usize>,
147
148    /// Specify the content-type header that should be sent back by the operation.
149    ///
150    /// This option is only meaningful when used along with presign.
151    pub override_content_type: Option<String>,
152    /// Specify the `cache-control` header that should be sent back by the operation.
153    ///
154    /// This option is only meaningful when used along with presign.
155    pub override_cache_control: Option<String>,
156    /// Specify the `content-disposition` header that should be sent back by the operation.
157    ///
158    /// This option is only meaningful when used along with presign.
159    pub override_content_disposition: Option<String>,
160}

300/// Options for write operations.
301#[derive(Debug, Clone, Default, Eq, PartialEq)]
302pub struct WriteOptions {
303    /// Sets append mode for this operation.
304    ///
305    /// ### Capability
306    ///
307    /// Check [`Capability::write_can_append`] before using this option.
308    ///
309    /// ### Behavior
310    ///
311    /// - By default, write operations overwrite existing files
312    /// - When append is set to true:
313    ///   - New data will be appended to the end of existing file
314    ///   - If file doesn't exist, it will be created
315    /// - If not supported, will return an error
316    ///
317    /// This operation allows adding data to existing files instead of overwriting them.
318    pub append: bool,
319
320    /// Sets Cache-Control header for this write operation.
321    ///
322    /// ### Capability
323    ///
324    /// Check [`Capability::write_with_cache_control`] before using this feature.
325    ///
326    /// ### Behavior
327    ///
328    /// - If supported, sets Cache-Control as system metadata on the target file
329    /// - The value should follow HTTP Cache-Control header format
330    /// - If not supported, the value will be ignored
331    ///
332    /// This operation allows controlling caching behavior for the written content.
333    ///
334    /// ### Use Cases
335    ///
336    /// - Setting browser cache duration
337    /// - Configuring CDN behavior
338    /// - Optimizing content delivery
339    /// - Managing cache invalidation
340    ///
341    /// ### References
342    ///
343    /// - [MDN Cache-Control](https://developer.mozilla.org/en-US/docs/Web/HTTP/Headers/Cache-Control)
344    /// - [RFC 7234 Section 5.2](https://tools.ietf.org/html/rfc7234#section-5.2)
345    pub cache_control: Option<String>,
346    /// Sets `Content-Type` header for this write operation.
347    ///
348    /// ## Capability
349    ///
350    /// Check [`Capability::write_with_content_type`] before using this feature.
351    ///
352    /// ### Behavior
353    ///
354    /// - If supported, sets Content-Type as system metadata on the target file
355    /// - The value should follow MIME type format (e.g. "text/plain", "image/jpeg")
356    /// - If not supported, the value will be ignored
357    ///
358    /// This operation allows specifying the media type of the content being written.
359    pub content_type: Option<String>,
360    /// Sets Content-Disposition header for this write request.
361    ///
362    /// ### Capability
363    ///
364    /// Check [`Capability::write_with_content_disposition`] before using this feature.
365    ///
366    /// ### Behavior
367    ///
368    /// - If supported, sets Content-Disposition as system metadata on the target file
369    /// - The value should follow HTTP Content-Disposition header format
370    /// - Common values include:
371    ///   - `inline` - Content displayed within browser
372    ///   - `attachment` - Content downloaded as file
373    ///   - `attachment; filename="example.jpg"` - Downloaded with specified filename
374    /// - If not supported, the value will be ignored
375    ///
376    /// This operation allows controlling how the content should be displayed or downloaded.
377    pub content_disposition: Option<String>,
378    /// Sets Content-Encoding header for this write request.
379    ///
380    /// ### Capability
381    ///
382    /// Check [`Capability::write_with_content_encoding`] before using this feature.
383    ///
384    /// ### Behavior
385    ///
386    /// - If supported, sets Content-Encoding as system metadata on the target file
387    /// - The value should follow HTTP Content-Encoding header format
388    /// - Common values include:
389    ///   - `gzip` - Content encoded using gzip compression
390    ///   - `deflate` - Content encoded using deflate compression
391    ///   - `br` - Content encoded using Brotli compression
392    ///   - `identity` - No encoding applied (default value)
393    /// - If not supported, the value will be ignored
394    ///
395    /// This operation allows specifying the encoding applied to the content being written.
396    pub content_encoding: Option<String>,
397    /// Sets user metadata for this write request.
398    ///
399    /// ### Capability
400    ///
401    /// Check [`Capability::write_with_user_metadata`] before using this feature.
402    ///
403    /// ### Behavior
404    ///
405    /// - If supported, the user metadata will be attached to the object during write
406    /// - Accepts key-value pairs where both key and value are strings
407    /// - Keys are case-insensitive in most services
408    /// - Services may have limitations for user metadata, for example:
409    ///   - Key length is typically limited (e.g., 1024 bytes)
410    ///   - Value length is typically limited (e.g., 4096 bytes)
411    ///   - Total metadata size might be limited
412    ///   - Some characters might be forbidden in keys
413    /// - If not supported, the metadata will be ignored
414    ///
415    /// User metadata provides a way to attach custom metadata to objects during write operations.
416    /// This metadata can be retrieved later when reading the object.
417    pub user_metadata: Option<HashMap<String, String>>,
418
419    /// Sets If-Match header for this write request.
420    ///
421    /// ### Capability
422    ///
423    /// Check [`Capability::write_with_if_match`] before using this feature.
424    ///
425    /// ### Behavior
426    ///
427    /// - If supported, the write operation will only succeed if the target's ETag matches the specified value
428    /// - The value should be a valid ETag string
429    /// - Common values include:
430    ///   - A specific ETag value like `"686897696a7c876b7e"`
431    ///   - `*` - Matches any existing resource
432    /// - If not supported, the value will be ignored
433    ///
434    /// This operation provides conditional write functionality based on ETag matching,
435    /// helping prevent unintended overwrites in concurrent scenarios.
436    pub if_match: Option<String>,
437    /// Sets If-None-Match header for this write request.
438    ///
439    /// Note: Certain services, like `s3`, support `if_not_exists` but not `if_none_match`.
440    /// Use `if_not_exists` if you only want to check whether a file exists.
441    ///
442    /// ### Capability
443    ///
444    /// Check [`Capability::write_with_if_none_match`] before using this feature.
445    ///
446    /// ### Behavior
447    ///
448    /// - If supported, the write operation will only succeed if the target's ETag does not match the specified value
449    /// - The value should be a valid ETag string
450    /// - Common values include:
451    ///   - A specific ETag value like `"686897696a7c876b7e"`
452    ///   - `*` - Matches if the resource does not exist
453    /// - If not supported, the value will be ignored
454    ///
455    /// This operation provides conditional write functionality based on ETag non-matching,
456    /// useful for preventing overwriting existing resources or ensuring unique writes.
457    pub if_none_match: Option<String>,
458    /// Sets the condition that write operation will succeed only if target does not exist.
459    ///
460    /// ### Capability
461    ///
462    /// Check [`Capability::write_with_if_not_exists`] before using this feature.
463    ///
464    /// ### Behavior
465    ///
466    /// - If supported, the write operation will only succeed if the target path does not exist
467    /// - Will return error if target already exists
468    /// - If not supported, the value will be ignored
469    ///
470    /// This operation provides a way to ensure write operations only create new resources
471    /// without overwriting existing ones, useful for implementing "create if not exists" logic.
472    pub if_not_exists: bool,
473
474    /// Sets concurrent write operations for this writer.
475    ///
476    /// ## Behavior
477    ///
478    /// - By default, OpenDAL writes files sequentially
479    /// - When concurrent is set:
480    ///   - Multiple write operations can execute in parallel
481    ///   - Write operations return immediately without waiting if tasks space are available
482    ///   - Close operation ensures all writes complete in order
483    ///   - Memory usage increases with concurrency level
484    /// - If not supported, falls back to sequential writes
485    ///
486    /// This feature significantly improves performance when:
487    /// - Writing large files
488    /// - Network latency is high
489    /// - Storage service supports concurrent uploads like multipart uploads
490    ///
491    /// ## Performance Impact
492    ///
493    /// Setting appropriate concurrency can:
494    /// - Increase write throughput
495    /// - Reduce total write time
496    /// - Better utilize available bandwidth
497    /// - Trade memory for performance
498    pub concurrent: usize,
499    /// Sets chunk size for buffered writes.
500    ///
501    /// ### Capability
502    ///
503    /// Check [`Capability::write_multi_min_size`] and [`Capability::write_multi_max_size`] for size limits.
504    ///
505    /// ### Behavior
506    ///
507    /// - By default, OpenDAL sets optimal chunk size based on service capabilities
508    /// - When chunk size is set:
509    ///   - Data will be buffered until reaching chunk size
510    ///   - One API call will be made per chunk
511    ///   - Last chunk may be smaller than chunk size
512    /// - Important considerations:
513    ///   - Some services require minimum chunk sizes (e.g. S3's EntityTooSmall error)
514    ///   - Smaller chunks increase API calls and costs
515    ///   - Larger chunks increase memory usage, but improve performance and reduce costs
516    ///
517    /// ### Performance Impact
518    ///
519    /// Setting appropriate chunk size can:
520    /// - Reduce number of API calls
521    /// - Improve overall throughput
522    /// - Lower operation costs
523    /// - Better utilize network bandwidth
524    pub chunk: Option<usize>,
525}
