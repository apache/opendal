23/// Metadata contains all the information related to a specific path.
24///
25/// Depending on the context of the requests, the metadata for the same path may vary. For example, two
26/// versions of the same path might have different content lengths. Keep in mind that metadata is always
27/// tied to the given context and is not a global state.
28///
29/// ## File Versions
30///
31/// In systems that support versioning, such as AWS S3, the metadata may represent a specific version
32/// of a file.
33///
34/// Users can access [`Metadata::version`] to retrieve the file's version, if available. They can also
35/// use [`Metadata::is_current`] and [`Metadata::is_deleted`] to determine whether the metadata
36/// corresponds to the latest version or a deleted one.
37///
38/// The all possible combinations of `is_current` and `is_deleted` are as follows:
39///
40/// | `is_current`  | `is_deleted` | description                                                                                                                                                                                                                                                                                                                                                                          |
41/// |---------------|--------------|--------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------|
42/// | `Some(true)`  | `false`      | **The metadata's associated version is the latest, current version.** This is the normal state, indicating that this version is the most up-to-date and accessible version.                                                                                                                                                                                                          |
43/// | `Some(true)`  | `true`       | **The metadata's associated version is the latest, deleted version (Latest Delete Marker or Soft Deleted).** This is particularly important in object storage systems like S3. It signifies that this version is the **most recent delete marker**, indicating the object has been deleted. Subsequent GET requests will return 404 errors unless a specific version ID is provided. |
44/// | `Some(false)` | `false`      | **The metadata's associated version is neither the latest version nor deleted.** This indicates that this version is a previous version, still accessible by specifying its version ID.                                                                                                                                                                                              |
45/// | `Some(false)` | `true`       | **The metadata's associated version is not the latest version and is deleted.** This represents a historical version that has been marked for deletion. Users will need to specify the version ID to access it, and accessing it may be subject to specific delete marker behavior (e.g., in S3, it might not return actual data but a specific delete marker response).             |
46/// | `None`        | `false`      | **The metadata's associated file is not deleted, but its version status is either unknown or it is not the latest version.** This likely indicates that versioning is not enabled for this file, or versioning information is unavailable.                                                                                                                                           |
47/// | `None`        | `true`       | **The metadata's associated file is deleted, but its version status is either unknown or it is not the latest version.** This typically means the file was deleted without versioning enabled, or its versioning information is unavailable. This may represent an actual data deletion operation rather than an S3 delete marker.                                                   |
48#[derive(Clone, Eq, PartialEq, Default)]
49pub struct Metadata {
50    mode: EntryMode,
51
52    is_current: Option<bool>,
53    is_deleted: bool,
54
55    cache_control: Option<String>,
56    content_disposition: Option<String>,
57    content_length: Option<u64>,
58    content_md5: Option<String>,
59    content_range: Option<BytesContentRange>,
60    content_type: Option<String>,
61    content_encoding: Option<String>,
62    etag: Option<String>,
63    last_modified: Option<Timestamp>,
64    version: Option<String>,
65
66    user_metadata: Option<HashMap<String, String>>,
67}
68
69impl fmt::Debug for Metadata {
70    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
71        let mut ds = f.debug_struct("Metadata");
72        ds.field("mode", &self.mode);
73
74        if let Some(is_current) = self.is_current {
75            ds.field("is_current", &is_current);
76        }
77        if self.is_deleted {
78            ds.field("is_deleted", &self.is_deleted);
79        }
80        if let Some(cache_control) = &self.cache_control {
81            ds.field("cache_control", cache_control);
82        }
83        if let Some(content_disposition) = &self.content_disposition {
84            ds.field("content_disposition", content_disposition);
85        }
86        if let Some(content_length) = self.content_length {
87            ds.field("content_length", &content_length);
88        }
89        if let Some(content_md5) = &self.content_md5 {
90            ds.field("content_md5", content_md5);
91        }
92        if let Some(content_range) = self.content_range {
93            ds.field("content_range", &content_range);
94        }
95        if let Some(content_type) = &self.content_type {
96            ds.field("content_type", content_type);
97        }
98        if let Some(content_encoding) = &self.content_encoding {
99            ds.field("content_encoding", content_encoding);
100        }
101        if let Some(etag) = &self.etag {
102            ds.field("etag", etag);
103        }
104        if let Some(last_modified) = self.last_modified {
105            ds.field("last_modified", &last_modified);
106        }
107        if let Some(version) = &self.version {
108            ds.field("version", version);
109        }
110        if let Some(user_metadata) = &self.user_metadata {
111            ds.field("user_metadata", user_metadata);
112        }
113
114        ds.finish()
115    }
116}

168    /// Checks whether the metadata corresponds to the most recent version of the file.
169    ///
170    /// This function is particularly useful when working with versioned objects,
171    /// such as those stored in systems like AWS S3 with versioning enabled. It helps
172    /// determine if the retrieved metadata represents the current state of the file
173    /// or an older version.
174    ///
175    /// Refer to docs in [`Metadata`] for more information about file versions.
176    ///
177    /// # Return Value
178    ///
179    /// The function returns an `Option<bool>` which can have the following values:
180    ///
181    /// - `Some(true)`:  Indicates that the metadata **is** associated with the latest version of the file.
182    ///   The metadata is current and reflects the most up-to-date state.
183    /// - `Some(false)`: Indicates that the metadata **is not** associated with the latest version of the file.
184    ///   The metadata belongs to an older version, and there might be a more recent version available.
185    /// - `None`:      Indicates that the currency of the metadata **cannot be determined**. This might occur if
186    ///   versioning is not supported or enabled, or if there is insufficient information to ascertain the version status.
187    pub fn is_current(&self) -> Option<bool> {
188        self.is_current
189    }
190
191    /// Set the `is_current` status of this entry.
192    ///
193    /// By default, this value will be `None`. Please avoid using this API if it's unclear whether the entry is current.
194    /// Set it to `true` if it is known to be the latest; otherwise, set it to `false`.
195    pub fn set_is_current(&mut self, is_current: bool) -> &mut Self {
196        self.is_current = Some(is_current);
197        self
198    }

490#[cfg(test)]
491mod tests {
492    use super::*;
493
494    #[test]
495    fn debug_metadata_omits_default_values() {
496        let metadata = Metadata::new(EntryMode::FILE);
497
498        assert_eq!(format!("{metadata:?}"), "Metadata { mode: FILE }");
499    }
500
501    #[test]
502    fn debug_metadata_keeps_meaningful_values() {
503        let metadata = Metadata::new(EntryMode::FILE)
504            .with_is_current(Some(false))
505            .with_is_deleted(true)
506            .with_content_length(42)
507            .with_version("v1".to_string());
508
509        assert_eq!(
510            format!("{metadata:?}"),
511            "Metadata { mode: FILE, is_current: false, is_deleted: true, content_length: 42, version: \"v1\" }"
512        );
513    }
